server {

        listen 80;
        server_name your.domain.com;
        root /var/www/enyphp/;
        index index.php index.html index.htm;

        try_files $uri $uri/ @rewrite;

        location @rewrite {
                rewrite ^/(.*)$ /index.php?/$1;
        }

        location ~ \.php$ {
                fastcgi_pass unix:/dev/shm/php-fpm.sock;
                include fastcgi_params;
                set $script "/index.php";
                set $pathinfo "";
                if ($uri ~* ^\/(.+?\.php)(/.*)$)
                {
                        set $script $1;
                        set $pathinfo $2;
                }
                fastcgi_param SCRIPT_NAME       	$script;
                fastcgi_param PATH_INFO         		$pathinfo;
                fastcgi_param SCRIPT_FILENAME  $document_root$script;
        }

        location ~* ^/(css|img|js|flv|swf|download|static)/(.+)$ {
				expire 30d; 
        }

        location ~ /\.ht {
                deny all;
        }

        access_log /var/log/nginx/your.domain.com_access_log;
        error_log /var/log/nginx/your.domain.com_error_log;
}
